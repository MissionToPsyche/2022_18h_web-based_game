1 /**
  2  * Class representing a Body
  3  * Defines the functionality for celestial bodies in the simulation
  4  * @extends Phaser.GameObjects.Sprite
  5  */
  6 class Body extends Phaser.GameObjects.Sprite {
  7 
  8 	/**
  9 	 * Represents a celestial body
 10 	 * @constructor
 11 	 * @param {Phaser.Scene} _scene - The Scene this Body will be added to
 12 	 * @param {Phaser.Geom.Point} _pos - The starting position of this body
 13 	 * @param {string} _id - The id of this Body
 14 	 * @param {number} _mass - The mass of this Body
 15 	 * @param {number} _diameter - The diameter of this Body
 16 	 * @param {Phaser.Textures.Frame} _frame - The frame this Body will be added to
 17 	 */
 18 	constructor (_scene, _pos, _id, _mass, _diameter, _frame) {
 19 		super(_scene, _pos.x, _pos.y, _id, _frame);
 20 		this.id = _id
 21 		this.mass = _mass
 22 		this.vel = new Phaser.Math.Vector2(0, 0)
 23 		this.r = _diameter / 2
 24 		this.listeners = []
 25 		this.listenRadius = 10 + this.r
 26 
 27 		this.setDisplaySize(this.r * 2, this.r * 2)
 28 			.setSize(this.r * 2, this.r *2);
 29 	}
 30 
 31 	/** Cause this body to orbit its parent if one exists and affect position by calculated velocity */
 32 	updatePosition(scene) {
 33 		if (this.parent != null) {
 34 			this.orbit(this.parent)
 35 		}
 36 
 37 		this.x += this.vel.x;
 38 		this.y += this.vel.y;
 39 	}
 40 
 41 	/** calculate velocity based off of force applied */
 42 	force(f) {
 43 		this.vel.add(gaussLaw(f, this.mass));
 44 	}
 45 
 46 	//begining of implementation of observer pattern to notify probes when close enough to annother body
 47 
 48 	/** add body to array of bodies that may be affected by dynamic gravity */
 49 	subscribe(listener) {
 50 		this.listeners.push(listener)
 51 	}
 52 
 53 	/** remove body from array of bodies that may be affected by dynamic gravity */
 54 	unsubscribe(listener) {
 55 		//yes, this is how you remove a specific array item in js. Yes, it's overly complicated.
 56 		this.listeners = this.listeners.filter(body => body.id != listener.id)
 57 	}
 58 
 59 	/** notify all subscribed listening bodies */
 60 	notify() {
 61 		if (!this.listeners) {
 62 			return
 63 		}
 64 
 65 		this.listeners.forEach(function (listener) {
 66 			//calculate radius from origin (this body) to lister
 67 			var p1 = new Phaser.Geom.Point(this.x, this.y);
 68 			var p2 = new Phaser.Geom.Point(listener.x, listener.y);
 69 			const r = Phaser.Math.Distance.BetweenPoints(p1, p2);
 70 
 71 			//check if lister is within listen radius
 72 			//also check if it's not within the planet so the probe isn't flung out of existance.
 73 			if (r <= this.listenRadius && r > this.r) {
 74 				//create vector f in direction of the listening body
 75                 const f = new Phaser.Math.Vector2(this.x - listener.x, this.y - listener.y);
 76 				//set direction vector to the length of the force applied by gravity between
 77 				//the two bodies, resulting in the force vector between the two bodies
 78 				f.setLength(calcGravity(listener.mass, this.mass, r));
 79 				//inform the listener of the force.
 80 				listener.update(f);
 81 			}
 82 		}.bind(this));
 83 	}
 84 
 85 	/** apply force from body subscribed to */
 86 	update(f) {
 87 		//NOTE** Might want to merge this function with this.force(f) at some point
 88 		this.vel.add(gaussLaw(f, this.mass).scale(0.1)); //TEMP divide by 2 'cause gravity too stronk
 89 	}
 90 }

 92 /**
 93  * Class representing a Satellite
 94  * Defines the functionality for a planet that orbit around the sun
 95  * @extends Body
 96  */
 97 class Satellite extends Body {
 98 
 99 	/**
100 	 * Represents a satellite, a type of body in the game that orbits around another body
101 	 * @constructor
102 	 * @param {Phaser.Scene} _scene - The Scene this Satellite will be added to
103 	 * @param {string} _id - The id of this Satellite
104 	 * @param {number} _mass - The mass of this Satellite
105 	 * @param {number} _diameter - The diameter of this Satellite
106 	 * @param {Body} _parent - The parent Body for this Satellite
107 	 * @param {number} _angle - The initial angle of the Satellite
108 	 * @param {number} _distance - The distance between the Satellite and its parent Body
109 	 * @param {Phaser.Textures.Frame} - The Frame this Satellite will be added to
110 	 */
111 	constructor (_scene, _id, _mass, _diameter, _parent, _angle, _distance, _frame) {
112 		super(_scene, CameraManager.getCenter(), _id, _mass, _diameter, _frame);
113 		this.scene = _scene;
114 		this.distance = _distance;
115 		this.path = [];
116 		this.angle = _angle;
117 		if (!this.angle) {
118 			this.angle = 0;
119 		}
120 
121 		if (_parent != null) {
122 			this.parent = _parent;
123 			this.x = this.parent.x;
124 			this.y = this.parent.y;
125 		}
126 
127 		this.x = this.x + this.distance * Math.cos(this.angle);
128 		this.y = this.y + this.distance * Math.sin(this.angle);
129 
130 		if (this.parent != null) {
131 			this.vel = orbitVelocity(this, this.parent, this.angle)
132 		}
133 	}
134 
135 	/**
136 	 * Return points on path as a curve
137 	 * @return {Phaser.Curves.Spline} Points on the path of this satellite
138 	 */
139 	getPathCurve () {
140 		return new Phaser.Curves.Spline(this.path);
141 	}
142 
143 	/** Add the current onscreen position into `this.path` */
144 	updatePosition(scene) {
145 		super.updatePosition(scene)
146 
147 		this.path.push(new Phaser.Math.Vector2(this.x, this.y));
148 		if (this.path.length > Math.min(this.mass * 10, (this.distance * Phaser.Math.PI2)/2)) {
149 			this.path.splice(0, 1)
150 		}
151 	}
152 
153 	/**
154 	 * Use Newton's law of Universal Gravitation to apply force to this Satellite and cause it to orbit
155 	 * its parent
156 	 */
157 	orbit(parent) {
158 		var p1 = new Phaser.Geom.Point(this.x, this.y);
159 		var p2 = new Phaser.Geom.Point(parent.x, parent.y);
160 		const r = Phaser.Math.Distance.BetweenPoints(p1, p2)
161 		const f = new Phaser.Math.Vector2(0, 0).copy(p2).subtract(p1)
162 
163 		// this is Newton's Law of Universal Gravitation (https://en.wikipedia.org/wiki/Newton%27s_law_of_universal_gravitation)
164 		// we use it here to calculate the force `parent` applies
165 		f.setLength(calcGravity(this.mass, parent.mass, r));
166 
167 		// and apply it
168 		this.force(f)
169 	}
170 }

172 /**
173  * Class representing a Moon
174  * Defines the functionality for a moon that orbit around a planet
175  * @extends Body
176  */
177 class Moon extends Body {
178 	/**
179 	 * Represents a moon, a type of body in the game that orbits around another body
180 	 * @constructor
181 	 * @param {Phaser.Scene} _scene - The Scene this Moon will be added to
182 	 * @param {string} _id - The id of this Moon
183 	 * @param {number} _mass - The mass of this Moon
184 	 * @param {number} _diameter - The diameter of this Moon
185 	 * @param {Body} _parent - The parent Body for this Moon
186 	 * @param {number} _angle - The initial angle of the Moon
187 	 * @param {number} _distance - The distance between the Moon and its parent Body
188 	 * @param {Phaser.Textures.Frame} - The Frame this Moon will be added to
189 	 */
190 	constructor (_scene, _id, _mass, _diameter, _parent, _angle, _distance, _frame) {
191 		super(_scene, CameraManager.getCenter(), _id, _mass, _diameter, _frame);
192 		this.scene = _scene
193 		this.parent = _parent;
194 		this.distance = _distance;
195 		this.path = [];
196 		this.theta = _angle;
197 		this.deltaTheta = 0.15;
198 
199 		if (this.parent != null) {
200 			this.parent = _parent;
201 		}
202 		// copy parent's position
203 		if (typeof(this.parent.pos) != "undefined" && this.parent.pos.x != 0) {
204 			this.x = this.parent.x + this.distance * Math.cos(this.theta);
205 			this.y = this.parent.y + this.distance * Math.sin(this.theta);
206 		}
207 	}
208 
209 	/**
210 	 * return points on path as a curve
211 	 * @return {Phaser.Curves.Spline} Points on the path of this moon
212 	 */
213 
214 	getPathCurve () {
215 		return new Phaser.Curves.Spline(this.path);
216 	}
217 
218 	/** Add the current onscreen position into `this.path` */
219 	updatePosition(scene) {
220 		if (typeof(this.parent) != "undefined" && this.parent.x != 0) {
221 			this.theta += this.deltaTheta;
222 			this.x = this.parent.x + this.distance * Math.cos(this.theta);
223 			this.y = this.parent.y + this.distance * Math.sin(this.theta);
224 		}
225 
226 		this.path.push(new Phaser.Math.Vector2(this.x, this.y));
227 		if (this.path.length > Math.min(this.mass * 10, (this.distance * Phaser.Math.PI2)/2)) {
228 			this.path.splice(0, 1)
229 		}
230 	}
231 }

233 /**
234  * Class representing a Probe
235  * Defines the functionality for a spacecraft
236  * @extends Body
237  */
238 class Probe extends Body {
239 
240 	/**
241 	 * Represents a probe, a type of body that the player can control
242 	 * @constructor
243 	 * @param {Phaser.Scene} _scene - The Scene this probe will be added to
244 	 * @param {string} _id - The id of this probe
245 	 * @param {number} _mass - The mass of this probe
246 	 * @param {number} _diameter - The diameter of this probe
247 	 * @param {Phaser.Textures.Frame} - The Frame this probe will be added to
248 	 */
249 	constructor (_scene, _id, _mass, _diameter, _frame) {
250 		super(_scene, CameraManager.getCenter(), _id, _mass, _diameter, _frame)
251 		// the initial state of gravity system
252 		// if the gravify is on at the beginning of the game, the Probe will have a 
253 		// initial velocity when starting from the earth
254 		this.gravityToggle = true; //TO DO: REMOVE WHEN DONE TESTING GRAVITY
255 
256 		this.x = this.scene.bodies["earth"].x;
257 		this.y = this.scene.bodies["earth"].y;
258 	}
259 
260 	/** Update the position of this probe */
261     update (f) {
262         //toggle for gravity
263 		//NOTE: FOR TESTING ONLY.
264 		if (!this.gravityToggle) {
265 			return
266 		}
267 
268         super.update(f);
269     }
270 
271     /**
272      * Get the distance from this probe to psyche
273      * @return {number} The distance between this probe and psyche
274      */
275     getPsycheDistance() {
276     	let psycheX = this.scene.bodies["psyche"].x;
277     	let psycheY = this.scene.bodies["psyche"].y;
278     	return Math.sqrt((this.x - psycheX) * (this.x - psycheX) + (this.y - psycheY) * (this.y - psycheY));
279     }
280 
281     /**
282      * Get the horizontal distance between the probe and psyche
283      * @return {number} the horizontal distancee between the probe and psyche
284      */
285     getPsycheDirectionX() {
286     	let psycheX = this.scene.bodies["psyche"].x;
287     	return (psycheX - this.x) / this.getPsycheDistance();
288     }
289 
290     /**
291      * Get the vertical distance between the probe and psyche
292      * @return {number} The vertical distance between the probe and psyche
293      */
294     getPsycheDirectionY() {
295     	let psycheY = this.scene.bodies["psyche"].y;
296     	return (psycheY - this.y) / this.getPsycheDistance();
297     }
