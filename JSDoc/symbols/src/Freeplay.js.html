1 /**
  2  * Class representing the Phaser 'Scene', which defines our game
  3  * @extends Phaser.Scene
  4  */
  5 class Freeplay extends Phaser.Scene {
  6 
  7     /**
  8      * Represents the scene and all of its variables
  9      * @constructor
 10      */
 11     constructor () {
 12         super({key:"Freeplay"});
 13         //creating body objects
 14         this.bodies = {};
 15         this.json;
 16         this.keyToggle = false //for testing only
 17         this.paused = false
 18         this.graphics;
 19         this.gravText;
 20         this.path;
 21         this.curve;
 22         this.points;
 23         this.graphics;
 24         this.pauseIndicator;
 25         this.direction;
 26     }
 27 
 28     /** Loads all necessary assets for the scene before the simulation runs */
 29     preload () {
 30         this.load.json('bodies', 'data/bodies.json');
 31 
 32         //loading in all image assets
 33         this.load.image('logo', 'img/Psyche_Icon_Color-SVG.svg'); //asset for psyche logo
 34         this.load.image('play', 'img/icons/play-circle.svg'); //asset for psyche logo
 35         this.load.image('pause', 'img/icons/pause-circle.svg'); //asset for psyche logo
 36         this.load.image('direction', 'img/icons/direction.png'); // an arrow
 37 
 38         //staticly loading all the individual assets for now
 39         //**TO DO: change to a more general method of preloading images
 40         this.load.image('earth', "img/icons/earth.svg");
 41         this.load.image('jupiter', "img/icons/jupiter.svg");
 42         this.load.image('luna', "img/icons/luna.svg");
 43         this.load.image('mars', "img/icons/mars.svg");
 44         this.load.image('mercury', "img/icons/mercury.svg");

 52         this.load.image('venus', "img/icons/venus.svg");
 53     }
 54 
 55     /**
 56      * Assembles the game within the scene
 57      * - Using information from data/bodies.json, Generate all bodies and add them to the scene
 58      * - Initialize the menu and game cameras
 59      * - Set the player as the probe
 60      * - Subscribe the probe to every body in the scene
 61      * - Generate UI sprites and add them to the scene
 62      * - Create player controls
 63      */
 64     create () {
 65         this.graphics = this.add.graphics();
 66 
 67         this.path = { t: 0, vec: new Phaser.Math.Vector2() };
 68 
 69         this.curve = new Phaser.Curves.Spline(this.points);
 70 
 71         //Solar system is 2048x2048
 72         this.matter.world.setBounds(0, 0, 2048, 2048);
 73 
 74         //initializing cameras
 75         CameraManager.initializeMainCamera(this);
 76         CameraManager.initializeUICamera(this);
 77 
 78         //creating Body objects
 79         this.json = this.cache.json.get('bodies');
 80         for (var type in this.json) {
 81             if (type != "moons") {
 82                 for (var body of this.json[type]) {
 83                     let id = body['id'];
 84                     let mass = body['mass']['value'];
 85                     let diameter = body['diameter']['value'];
 86 
 87                     //objects in group 1 (in this case Satellites) will not collide with each other
 88                     let collisionGroup1 = this.matter.world.nextGroup(true);
 89                     let collisionGroup2 = this.matter.world.nextGroup();
 90 
 91                     if(type != "probes"){
 92                         let parent = this.bodies[body['orbits']];
 93                         let angle = body['angle'];
 94                         let orbit_distance = body['orbit_distance']['value'];
 95                         this.bodies[id] = new Satellite(this, id, mass, diameter, parent, angle, orbit_distance);
 96                     } else {
 97                         this.bodies[id] = new Probe(this, id, mass, diameter);
 98                     }
 99                 }
100             } else {
101                 // create satellites such as luna
102                 for (var body of this.json[type]) {
103                     let id = body['id'];
104                     let mass = body['mass']['value'];
105                     let diameter = body['diameter']['value'];
106                     let parent = this.bodies[body['orbits']];
107                     let angle = body['angle'];
108                     let orbit_distance = body['orbit_distance']['value'];
109                     this.bodies[id] = new Moon(this, id, mass, diameter, parent, angle, orbit_distance);
110                 }
111             }
112         }
113 
114         for (const body in this.bodies) {
115             //add each body to the scene
116             this.add.existing(this.bodies[body]);
117             //add bodies to game sprites so that they don't
118             //appear on UI camera
119             CameraManager.addGameSprite(this.bodies[body]);
120         }
121         CameraManager.addGameSprite(this.graphics); //adding graphics to game sprites so that it doesn't show up in UI.
122 
123         //subscribe probe to all other bodies.
124         //NOTE** hard coded to psyche probe for now
125         for (const body in this.bodies) {
126             if(this.bodies[body].id != "psyche_probe"){
127                 this.bodies[body].subscribe(this.bodies["psyche_probe"]);
128             }
129         }
130         this.bodies["earth"].subscribe(this.bodies["luna"]);
131         //setting probe as the player
132         this.player = this.bodies["psyche_probe"];
133         CameraManager.setFollowSprite(this.player);
134 
135         //creating UISprites
136         var logo = this.add.image(50,50,'logo').setScale(0.5);
137         this.gravText = this.add.text(4, 90, '0')
138         this.gravText.setText("Gravity: ON")
139         this.playIndicator = this.add.image(964, 708, 'play').setScale(0.5)
140         this.pauseIndicator = this.add.image(964, 708, 'pause').setScale(0.5)
141 
142         //adding to UIsprites so main camera ignores them
143         CameraManager.addUISprite(logo);
144         CameraManager.addUISprite(this.gravText);
145         CameraManager.addUISprite(this.playIndicator);
146         CameraManager.addUISprite(this.pauseIndicator);
147 
148         //creating control keys
149         this.cursors = this.input.keyboard.createCursorKeys();
150         this.spaceKey = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.SPACE);
151         this.pauseKey = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.P);
152 
153         this.input.on("pointerdown", function (pointer){
154             if ((pointer.x > 934) && (pointer.x < 994) && (pointer.y > 678) &&(pointer.y < 738)) {
155                 this.paused = !this.paused;
156             }
157         }, this);
158         console.log(Phaser.Input.Keyboard.SPACEBAR)
159     }
160 
161     /** The scene's main update loop
162      * - Disallows the probe from escaping the solar system or going to fast
163      * - Applies dynamic gravity
164      * - Enforces the pause feature, only allowing bodies to move if the game is not paused
165      */
166     update () {
167         //Probe controls
168         //**TO DO: Wrap in a custom controler later.
169         const moveUnit = 0.01
170 
171         // update pause/play indicator
172         if (this.paused) {
173             this.playIndicator.setVisible(false)
174             this.pauseIndicator.setVisible(true)
175         } else {
176             this.pauseIndicator.setVisible(false)
177             this.playIndicator.setVisible(true)
178         }
179 
180         // only move if not paused
181         if (!this.paused) {
182             if (this.cursors.left.isDown) {
183                 this.bodies["psyche_probe"].vel.x -= moveUnit;
184             }
185             else if (this.cursors.right.isDown) {
186                 this.bodies["psyche_probe"].vel.x += moveUnit;
187             }
188 
189             if (this.cursors.up.isDown) {
190                 this.bodies["psyche_probe"].vel.y -= moveUnit;
191             }
192             else if (this.cursors.down.isDown) {
193                 this.bodies["psyche_probe"].vel.y += moveUnit;
194             }
195         }
196 
197         if (this.spaceKey.isDown) {
198             this.bodies["psyche_probe"].gravityToggle = !this.keyToggle ? !this.bodies["psyche_probe"].gravityToggle : this.bodies["psyche_probe"].gravityToggle
199             this.gravText.setText("Gravity: " + (this.bodies["psyche_probe"].gravityToggle ? "ON" : "OFF"))
200             this.keyToggle = true
201         } else if (this.pauseKey.isDown) {
202             this.paused = !this.keyToggle ? !this.paused : this.paused
203             this.keyToggle = true
204         } else {
205             this.keyToggle = false
206         }
207 
208         //prevent psyche from going too far out for now
209 	    //note: FOR TESTING ONLY, THIS IS A BAD WAY OF DOING THIS
210         if (this.bodies["psyche_probe"].x >= 650 + 1024) {
211             this.bodies["psyche_probe"].vel.x = 0
212             this.bodies["psyche_probe"].x = 649 + 1024
213         } if (this.bodies["psyche_probe"].y >= 650 + 1024) {
214             this.bodies["psyche_probe"].vel.y = 0
215             this.bodies["psyche_probe"].y = 649 + 1024
216         } if (this.bodies["psyche_probe"].x <= -650 + 1024) {
217             this.bodies["psyche_probe"].vel.x = 0
218             this.bodies["psyche_probe"].x = -649 + 1024
219         } if (this.bodies["psyche_probe"].y <= -650 + 1024) {
220             this.bodies["psyche_probe"].vel.y = 0
221             this.bodies["psyche_probe"].y = -649 + 1024
222         }
223 
224         // don't update bodies if paused
225         if (this.paused) {
226             return
227         }
228 
229         this.graphics.clear(); //clear previous itteration's graphics
230 
231         for (const body in this.bodies) {
232             //apply dynamic gravity
233             //NOTE: THIS IS A BAD PLACE TO DO THIS. MOVE THIS TO AN APPROPRIATE PLACE LATER!!
234             this.bodies[body].notify() 
235 
236             //draw paths
237             var path = this.bodies[body].path;
238             if(path && path.length > 0){
239                 this.graphics.lineStyle(1, 0xffffff, 0.5);
240                 this.bodies[body].getPathCurve().draw(this.graphics, 64);
241         
242                 this.graphics.fillStyle(0x00ff00, 1);
243             }
244     
245             //update body positions
246             this.bodies[body].updatePosition(this)
247 
248             // find psyche
249             let distance = this.bodies["psyche_probe"].getPsycheDistance();
250 
251             // the distance between pshche probe and the arrow
252             let arrowDistance = 50;
253             let width = 1024;
254             let height = 768;
255             let directionX = width / 2 + this.bodies["psyche_probe"].getPsycheDirectionX() * arrowDistance;
256             let directionY = height / 2 + this.bodies["psyche_probe"].getPsycheDirectionY() * arrowDistance;
257             
258             // calculate the rotation of the arrow image
259             let directionAngle = Math.asin(this.bodies["psyche_probe"].getPsycheDirectionY());
260             if (this.bodies["psyche_probe"].getPsycheDirectionX() < 0) {
261                 directionAngle = Math.PI - directionAngle;
262             }
263 
264             // add the image of the arrow if it not added
265             if (typeof(this.direction) == "undefined") {
266                 this.direction = this.add.image(directionX, directionY, 'direction').setScale(0.2);
267                 CameraManager.addUISprite(this.direction);
268             }
269 
270             // set the correct position and angle of the arrow to point to psyche
271             this.direction.setPosition(directionX, directionY);
272             this.direction.rotation = directionAngle;
273 
274             // decrease opacity when near psyche
275             if (distance < 90) {
276                 this.direction.alpha = (distance - 50)/50;
277             } else {
278                 this.direction.alpha = 0.8;
279             }
280         }
281     }
